51#[inline]
52fn tau(a: u32) -> u32 {
53    let mut buf = [0u8; 4];
54    BE::write_u32(&mut buf, a);
55    buf[0] = SBOX[buf[0] as usize];
56    buf[1] = SBOX[buf[1] as usize];
57    buf[2] = SBOX[buf[2] as usize];
58    buf[3] = SBOX[buf[3] as usize];
59    BE::read_u32(&buf)
60}

62/// L: linear transformation
63#[inline]
64fn el(b: u32) -> u32 {
65    b ^ b.rotate_left(2) ^ b.rotate_left(10) ^ b.rotate_left(18) ^ b.rotate_left(24)
66}
67
68#[inline]
69fn el_prime(b: u32) -> u32 {
70    b ^ b.rotate_left(13) ^ b.rotate_left(23)
71}
72
73#[inline]
74fn t(val: u32) -> u32 {
75    el(tau(val))
76}
77
78#[inline]
79fn t_prime(val: u32) -> u32 {
80    el_prime(tau(val))
81}

83/// SM4 cipher
84#[derive(Copy, Clone)]
85pub struct Sm4 {
86    rk: [u32; 32],
87}
88
89impl NewBlockCipher for Sm4 {
90    type KeySize = U16;
91
92    fn new(key: &GenericArray<u8, U16>) -> Self {
93        let mut mk = [0u32; 4];
94        let mut rk = [0u32; 32];
95        BE::read_u32_into(key, &mut mk);
96        let mut k = [mk[0] ^ FK[0], mk[1] ^ FK[1], mk[2] ^ FK[2], mk[3] ^ FK[3]];
97
98        for i in 0..8 {
99            k[0] ^= t_prime(k[1] ^ k[2] ^ k[3] ^ CK[i * 4]);
100            k[1] ^= t_prime(k[2] ^ k[3] ^ k[0] ^ CK[i * 4 + 1]);
101            k[2] ^= t_prime(k[3] ^ k[0] ^ k[1] ^ CK[i * 4 + 2]);
102            k[3] ^= t_prime(k[0] ^ k[1] ^ k[2] ^ CK[i * 4 + 3]);
103
104            rk[i * 4] = k[0];
105            rk[i * 4 + 1] = k[1];
106            rk[i * 4 + 2] = k[2];
107            rk[i * 4 + 3] = k[3];
108        }
109
110        Sm4 { rk }
111    }
112}

119impl BlockEncrypt for Sm4 {
120    fn encrypt_block(&self, block: &mut GenericArray<u8, U16>) {
121        let mut x = [0u32; 4];
122        BE::read_u32_into(block, &mut x);
123        let rk = &self.rk;
124        for i in 0..8 {
125            x[0] ^= t(x[1] ^ x[2] ^ x[3] ^ rk[i * 4]);
126            x[1] ^= t(x[2] ^ x[3] ^ x[0] ^ rk[i * 4 + 1]);
127            x[2] ^= t(x[3] ^ x[0] ^ x[1] ^ rk[i * 4 + 2]);
128            x[3] ^= t(x[0] ^ x[1] ^ x[2] ^ rk[i * 4 + 3]);
129        }
130        x = [x[3], x[2], x[1], x[0]];
131        BE::write_u32_into(&x, block);
132    }
133}

135impl BlockDecrypt for Sm4 {
136    fn decrypt_block(&self, block: &mut GenericArray<u8, U16>) {
137        let mut x = [0u32; 4];
138        BE::read_u32_into(block, &mut x);
139        let rk = &self.rk;
140        for i in 0..8 {
141            x[0] ^= t(x[1] ^ x[2] ^ x[3] ^ rk[31 - i * 4]);
142            x[1] ^= t(x[2] ^ x[3] ^ x[0] ^ rk[31 - (i * 4 + 1)]);
143            x[2] ^= t(x[3] ^ x[0] ^ x[1] ^ rk[31 - (i * 4 + 2)]);
144            x[3] ^= t(x[0] ^ x[1] ^ x[2] ^ rk[31 - (i * 4 + 3)]);
145        }
146        x = [x[3], x[2], x[1], x[0]];
147        BE::write_u32_into(&x, block);
148    }
